2
हरेक HTML Page मा Script Tags थप्नुस्

हरेक HTML file को closing </body> tag भन्दा अगाडि यो code थप्नुस्:

<!-- Firebase Scripts — सबै page मा थप्नुस् -->
<script type="module">
  import { openLoginModal, guildSignOut }
    from './js/guild-auth.js';
  import { renderLeaderboardIntoDOM }
    './js/guild-db.js';

  // Login button wiring
  document.querySelectorAll('[data-auth="login-btn"]')
    .forEach(btn => btn.onclick = openLoginModal);

  // Live leaderboard (if element exists on this page)
  renderLeaderboardIntoDOM('leaderboard-list');
</script>
✅ type="module" → Firebase SDK ko lagi required. Sab kaam garcha.

5
Firestore Security Rules — Copy गर्नुस्

Firestore → Rules tab मा जानुस् र यो paste गर्नुस्:

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Players — read public, write own only
    match /players/{uid} {
      allow read:  if true;
      allow write: if request.auth.uid == uid;
    }

    // Posts — read public, write creators only
    match /posts/{postId} {
      allow read:   if true;
      allow create: if request.auth != null
        && get(/databases/$(database)/documents/players/$(request.auth.uid))
           .data.isCreator == true;
      allow update,delete: if
        request.auth.uid == resource.data.authorId;
    }

    // Quest Results — write own, read own
    match /quest_results/{resultId} {
      allow read,write: if
        request.auth.uid == request.resource.data.userId;
    }

    // Gallery — read public, write logged-in
    match /gallery/{imageId} {
      allow read:   if true;
      allow create: if request.auth != null;
      allow delete: if
        request.auth.uid == resource.data.uid;
    }

    // Creator Apps — write own, read admin only
    match /creator_apps/{appId} {
      allow create: if request.auth != null;
      allow read,update: if
        request.auth.uid == resource.data.uid;
    }

    // Quests — read only
    match /quests/{questId} {
      allow read: if true;
      allow write: if false; // Admin only via console
      match /questions/{qId} {
        allow read: if true;
      }
    }
  }
}
